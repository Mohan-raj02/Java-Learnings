// static method cannot be overrides --(example)

class Parent {
    static void show() {
        System.out.println("Static method in Parent");
    }
}

class Child extends Parent {
    // @Override // This is not overriding, but hiding the static method  // cant use'override annotation when says ovverriding method'
    static void show() {
        System.out.println("Static method in Child");
    }
}

public class Test {
    public static void main(String[] args) {
        Parent p = new Parent();
        p.show(); // Calls Parent.show()

        Child c = new Child();
        c.show(); // Calls Child.show()

        Parent pc = new Child();
        pc.show(); // Still calls Parent.show(), not Child.show()
    }
}
